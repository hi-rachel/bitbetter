```python
def lengthOfLongestSubstring_bruteforce(s: str) -> int:
    def is_unique(start, end):
        chars = set()
        for i in range(start, end + 1):
            if s[i] in chars:
                return False
            chars.add(s[i])
        return True

    n = len(s)
    max_length = 0

    for i in range(n):
        for j in range(i, n):
            if is_unique(i, j):
                max_length = max(max_length, j - i + 1)

    return max_length
```
